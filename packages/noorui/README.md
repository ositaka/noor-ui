## Installation

```bash
npm install noorui-rtl
```

## Quick Start

```tsx
import { NoorProvider } from 'noorui-rtl';
import 'noorui-rtl/styles.css';

function App() {
  return (
    <NoorProvider locale="ar" direction="rtl">
      <YourApp />
    </NoorProvider>
  );
}
```
